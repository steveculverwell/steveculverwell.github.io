Reduce

example taken from Fun Fun Functions

const runs = [
  {distance: 5},
  {distance: 10},
  {distance: 21},
  {distance: 42},
  {distance: 100}
]

//@sum is the value of each iteration adding on to it each time.
//@run is each iteration or loop over the runs array
const totalDistance = runs.reduce((sum, run) => sum + run.distance, 0);

console.log(`You've run ${totalDistance}km`);
