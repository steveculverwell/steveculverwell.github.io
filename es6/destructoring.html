Destructuring

example taken from http://wesbos.com/courses

Allows you to extract data items from an array into their own variable

const triathlon = {
	first: 'alistar',
	last: 'brownlee',
	country: 'england',
	medals: {
		superleague: 'no',
		olympic: 'gold',
		itu: 'gold'
	},
	sponsor: {
		one: 'aldi',
		two: 'yorkshire tea',
		three: 'scott bikes',
		four: 'huub'
	}

}

const first_ = triathlon.first;
const last_ = triathlon.last;
//console.log(first_, last_);

//give me a var called first and last and take it from triathlon
const { first, last } = triathlon;
const { olympic, itu } = triathlon.medals;
